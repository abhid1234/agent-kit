```typescript
// src/strategy/sequential.ts
import type { Strategy, TeamResult, StrategyOptions, AgentResponse } from './interface';
import type { Agent } from '../agent';
import type { AgentEventEmitter } from '../events';

export class SequentialStrategy implements Strategy {
  async execute(
    agents: Agent[],
    task: string,
    _options: StrategyOptions,
    emitter: AgentEventEmitter,
  ): Promise<TeamResult> {
    const responses: AgentResponse[] = [];
    let previousOutput = '';

    for (const agent of agents) {
      const agentTask = previousOutput
        ? `Previous agent produced:\n${previousOutput}\n\nNow it's your turn. ${task}`
        : task;

      emitter.emit({
        type: 'team:agent:start',
        timestamp: Date.now(),
        agentId: agent.name,
        data: { agent: agent.name, task: agentTask },
      });

      const startTime = Date.now();
      const result = await agent.chat(agentTask);
      const latencyMs = Date.now() - startTime;

      previousOutput = result.content;
      responses.push({ agent: agent.name, content: result.content });

      emitter.emit({
        type: 'team:agent:end',
        timestamp: Date.now(),
        agentId: agent.name,
        data: { agent: agent.name, content: result.content },
        latencyMs,
      });
    }

    return {
      content: previousOutput,
      responses,
    };
  }
}
```

```typescript
// src/strategy/parallel.ts
import type { Strategy, TeamResult, StrategyOptions, AgentResponse } from './interface';
import type { Agent } from '../agent';
import type { AgentEventEmitter } from '../events';

export class ParallelStrategy implements Strategy {
  async execute(
    agents: Agent[],
    task: string,
    _options: StrategyOptions,
    emitter: AgentEventEmitter,
  ): Promise<TeamResult> {
    const promises = agents.map(async (agent) => {
      emitter.emit({
        type: 'team:agent:start',
        timestamp: Date.now(),
        agentId: agent.name,
        data: { agent: agent.name, task },
      });

      const startTime = Date.now();
      const result = await agent.chat(task);
      const latencyMs = Date.now() - startTime;

      emitter.emit({
        type: 'team:agent:end',
        timestamp: Date.now(),
        agentId: agent.name,
        data: { agent: agent.name, content: result.content },
        latencyMs,
      });

      return { agent: agent.name, content: result.content } as AgentResponse;
    });

    const responses = await Promise.all(promises);

    const content = responses
      .map((r) => `[${r.agent}]: ${r.content}`)
      .join('\n\n');

    return { content, responses };
  }
}
```

```typescript
// src/strategy/debate.ts
import type { Strategy, TeamResult, StrategyOptions, AgentResponse } from './interface';
import type { Agent } from '../agent';
import type { AgentEventEmitter } from '../events';

export class DebateStrategy implements Strategy {
  async execute(
    agents: Agent[],
    task: string,
    options: StrategyOptions,
    emitter: AgentEventEmitter,
  ): Promise<TeamResult> {
    const maxRounds = options.maxRounds ?? 3;
    const responses: AgentResponse[] = [];
    let conversationHistory = '';

    for (let round = 1; round <= maxRounds; round++) {
      for (const agent of agents) {
        const prompt = round === 1 && agent === agents[0]
          ? task
          : `Original task: ${task}\n\nConversation so far:\n${conversationHistory}\n\nProvide your response.`;

        emitter.emit({
          type: 'team:agent:start',
          timestamp: Date.now(),
          agentId: agent.name,
          data: { agent: agent.name, task: prompt, round },
        });

        const startTime = Date.now();
        const result = await agent.chat(prompt);
        const latencyMs = Date.now() - startTime;

        conversationHistory += `\n[${agent.name}, round ${round}]: ${result.content}`;
        responses.push({ agent: agent.name, content: result.content, round });

        emitter.emit({
          type: 'team:agent:end',
          timestamp: Date.now(),
          agentId: agent.name,
          data: { agent: agent.name, content: result.content },
          latencyMs,
        });
      }

      emitter.emit({
        type: 'team:round',
        timestamp: Date.now(),
        agentId: 'team',
        data: { round, responses: responses.filter((r) => r.round === round) },
      });
    }

    // Final answer from the first agent
    const finalPrompt = `Original task: ${task}\n\nDebate history:\n${conversationHistory}\n\nBased on this debate, provide your final answer.`;

    const startTime = Date.now();
    const finalResult = await agents[0].chat(finalPrompt);

    emitter.emit({
      type: 'team:agent:end',
      timestamp: Date.now(),
      agentId: agents[0].name,
      data: { agent: agents[0].name, content: finalResult.content },
      latencyMs: Date.now() - startTime,
    });

    return {
      content: finalResult.content,
      responses,
    };
  }
}
```

- [ ] **Step 3: Implement HierarchicalStrategy**

This is the most complex strategy. The manager gets an auto-injected `delegate` tool. The strategy handles the delegation loop.

```typescript
// src/strategy/hierarchical.ts
import type { Strategy, TeamResult, StrategyOptions, AgentResponse } from './interface';
import type { Agent } from '../agent';
import type { AgentEventEmitter } from '../events';
import { Tool } from '../tool';
import { createMessage, isToolCallMessage } from '../types';
import type { ModelAdapter } from '../model/adapter';
import type { Message, ToolDefinition } from '../types';

export class HierarchicalStrategy implements Strategy {
  async execute(
    agents: Agent[],
    task: string,
    options: StrategyOptions,
    emitter: AgentEventEmitter,
  ): Promise<TeamResult> {
    const manager = options.manager;
    if (!manager) {
      throw new Error('HierarchicalStrategy requires a manager agent');
    }

    const maxDelegations = options.maxDelegations ?? 10;
    const agentMap = new Map(agents.map((a) => [a.name, a]));
    const responses: AgentResponse[] = [];
    let delegationCount = 0;

    // Build the delegate tool definition
    const delegateTool: ToolDefinition = {
      name: 'delegate',
      description: 'Delegate a task to a team member. Available agents: ' + agents.map((a) => a.name).join(', '),
      parameters: {
        agent: { type: 'string', description: 'Name of the agent to delegate to' },
        task: { type: 'string', description: 'What to ask this agent to do' },
      },
    };

    // We need to manually drive the manager's chat loop because we
    // intercept the delegate tool call ourselves
    const messages: Message[] = [
      createMessage({
        role: 'system',
        content: `You are a manager coordinating a team. Available team members: ${agents.map((a) => a.name).join(', ')}. Use the delegate tool to assign tasks to team members. When you have enough information, respond directly with the final answer.`,
      }),
      createMessage({ role: 'user', content: task }),
    ];

    // Access manager's model adapter — we need to drive the loop ourselves
    // Since Agent doesn't expose its model, we use manager.chat() but need
    // to intercept tool calls. The simplest approach: give manager the delegate
    // tool and let Agent handle the tool execution loop.

    // Create a wrapper that executes delegations
    const delegateToolInstance = Tool.create({
      name: 'delegate',
      description: delegateTool.description,
      parameters: delegateTool.parameters,
      execute: async (params) => {
        const agentName = params.agent as string;
        const agentTask = params.task as string;
        const targetAgent = agentMap.get(agentName);

        if (!targetAgent) {
          return `Error: agent "${agentName}" not found. Available: ${agents.map((a) => a.name).join(', ')}`;
        }

        delegationCount++;

        emitter.emit({
          type: 'team:delegate',
          timestamp: Date.now(),
          agentId: manager.name,
          data: { from: manager.name, to: agentName, task: agentTask },
        });

        emitter.emit({
          type: 'team:agent:start',
          timestamp: Date.now(),
          agentId: agentName,
          data: { agent: agentName, task: agentTask },
        });

        const startTime = Date.now();
        const result = await targetAgent.chat(agentTask);
        const latencyMs = Date.now() - startTime;

        responses.push({ agent: agentName, content: result.content });

        emitter.emit({
          type: 'team:agent:end',
          timestamp: Date.now(),
          agentId: agentName,
          data: { agent: agentName, content: result.content },
          latencyMs,
        });

        return result.content;
      },
    });

    // Create a manager agent clone with the delegate tool added
    // Since we can't modify the existing agent, create a new one with same config + delegate tool
    const managerWithDelegate = new Agent({
      name: manager.name,
      model: (manager as any).model ?? undefined,
      system: `You are a manager coordinating a team. Available team members: ${agents.map((a) => a.name).join(', ')}. Use the delegate tool to assign tasks to team members. When you have enough information, respond directly with the final answer.`,
      tools: [delegateToolInstance],
      maxToolRounds: maxDelegations,
    });

    emitter.emit({
      type: 'team:agent:start',
      timestamp: Date.now(),
      agentId: manager.name,
      data: { agent: manager.name, task },
    });

    const startTime = Date.now();
    const result = await managerWithDelegate.chat(task);

    emitter.emit({
      type: 'team:agent:end',
      timestamp: Date.now(),
      agentId: manager.name,
      data: { agent: manager.name, content: result.content },
      latencyMs: Date.now() - startTime,
    });

    return {
      content: result.content,
      responses,
    };
  }
}
```

```typescript
// src/team.ts
import type { Agent } from './agent';
import { AgentEventEmitter } from './events';
import type { AgentEvent } from './types';
import type { Strategy, TeamResult, StrategyOptions } from './strategy/interface';
import { SequentialStrategy } from './strategy/sequential';
import { ParallelStrategy } from './strategy/parallel';
import { DebateStrategy } from './strategy/debate';
import { HierarchicalStrategy } from './strategy/hierarchical';

export type { TeamResult, AgentResponse } from './strategy/interface';

export interface TeamConfig {
  agents: Agent[];
  strategy: 'sequential' | 'parallel' | 'debate' | 'hierarchical';
  manager?: Agent;
  maxRounds?: number;
  maxDelegations?: number;
}

const strategies: Record<string, Strategy> = {
  sequential: new SequentialStrategy(),
  parallel: new ParallelStrategy(),
  debate: new DebateStrategy(),
  hierarchical: new HierarchicalStrategy(),
};

export class Team {
  private agents: Agent[];
  private strategy: Strategy;
  private strategyName: string;
  private options: StrategyOptions;
  private emitter = new AgentEventEmitter();

  constructor(config: TeamConfig) {
    this.agents = config.agents;
    this.strategyName = config.strategy;
    this.strategy = strategies[config.strategy];
    this.options = {
      manager: config.manager,
      maxRounds: config.maxRounds,
      maxDelegations: config.maxDelegations,
    };
  }

  on(type: string, handler: (event: AgentEvent) => void): void {
    this.emitter.on(type, handler);
  }

  off(type: string, handler: (event: AgentEvent) => void): void {
    this.emitter.off(type, handler);
  }

  async run(task: string): Promise<TeamResult> {
    this.emitter.emit({
      type: 'team:start',
      timestamp: Date.now(),
      agentId: 'team',
      data: {
        task,
        strategy: this.strategyName,
        agents: this.agents.map((a) => a.name),
      },
    });

    const startTime = Date.now();
    const result = await this.strategy.execute(this.agents, task, this.options, this.emitter);

    this.emitter.emit({
      type: 'team:end',
      timestamp: Date.now(),
      agentId: 'team',
      data: { content: result.content },
      latencyMs: Date.now() - startTime,
    });

    return result;
  }
}
```
